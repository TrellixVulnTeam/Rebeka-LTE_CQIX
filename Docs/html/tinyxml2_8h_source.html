198 /*
199  A dynamic array of Plain Old Data. Doesn't support constructors, etc.
200  Has a small initial memory pool, so that low or no usage will not
201  cause a call to new/delete
202 */
203 template <class T, int INIT>
204 class DynArray
205 {
206 public:
207  DynArray< T, INIT >() {
208  _mem = _pool;
209  _allocated = INIT;
210  _size = 0;
211  }
212 
213  ~DynArray() {
214  if ( _mem != _pool ) {
215  delete [] _mem;
216  }
217  }
218 
219  void Clear() {
220  _size = 0;
221  }
222 
223  void Push( T t ) {
224  EnsureCapacity( _size+1 );
225  _mem[_size++] = t;
226  }
227 
228  T* PushArr( int count ) {
229  EnsureCapacity( _size+count );
230  T* ret = &_mem[_size];
231  _size += count;
232  return ret;
233  }
234 
235  T Pop() {
236  return _mem[--_size];
237  }
238 
239  void PopArr( int count ) {
240  TIXMLASSERT( _size >= count );
241  _size -= count;
242  }
243 
244  bool Empty() const {
245  return _size == 0;
246  }
247 
248  T& operator[](int i) {
249  TIXMLASSERT( i>= 0 && i < _size );
250  return _mem[i];
251  }
252 
253  const T& operator[](int i) const {
254  TIXMLASSERT( i>= 0 && i < _size );
255  return _mem[i];
256  }
257 
258  const T& PeekTop() const {
259  TIXMLASSERT( _size > 0 );
260  return _mem[ _size - 1];
261  }
262 
263  int Size() const {
264  return _size;
265  }
266 
267  int Capacity() const {
268  return _allocated;
269  }
270 
271  const T* Mem() const {
272  return _mem;
273  }
274 
275  T* Mem() {
276  return _mem;
277  }
278 
279 private:
280  void EnsureCapacity( int cap ) {
281  if ( cap > _allocated ) {
282  int newAllocated = cap * 2;
283  T* newMem = new T[newAllocated];
284  memcpy( newMem, _mem, sizeof(T)*_size ); // warning: not using constructors, only works for PODs
285  if ( _mem != _pool ) {
286  delete [] _mem;
287  }
288  _mem = newMem;
289  _allocated = newAllocated;
290  }
291  }
292 
293  T* _mem;
294  T _pool[INIT];
295  int _allocated; // objects allocated
296  int _size; // number objects in use
297 };

317 /*
318  Template child class to create pools of the correct type.
319 */
320 template< int SIZE >
321 class MemPoolT : public MemPool
322 {
323 public:
324  MemPoolT() : _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0), _nUntracked(0) {}
325  ~MemPoolT() {
326  // Delete the blocks.
327  for( int i=0; i<_blockPtrs.Size(); ++i ) {
328  delete _blockPtrs[i];
329  }
330  }
331 
332  virtual int ItemSize() const {
333  return SIZE;
334  }
335  int CurrentAllocs() const {
336  return _currentAllocs;
337  }
338 
339  virtual void* Alloc() {
340  if ( !_root ) {
341  // Need a new block.
342  Block* block = new Block();
343  _blockPtrs.Push( block );
344 
345  for( int i=0; i<COUNT-1; ++i ) {
346  block->chunk[i].next = &block->chunk[i+1];
347  }
348  block->chunk[COUNT-1].next = 0;
349  _root = block->chunk;
350  }
351  void* result = _root;
352  _root = _root->next;
353 
354  ++_currentAllocs;
355  if ( _currentAllocs > _maxAllocs ) {
356  _maxAllocs = _currentAllocs;
357  }
358  _nAllocs++;
359  _nUntracked++;
360  return result;
361  }
362  virtual void Free( void* mem ) {
363  if ( !mem ) {
364  return;
365  }
366  --_currentAllocs;
367  Chunk* chunk = (Chunk*)mem;
368 #ifdef DEBUG
369  memset( chunk, 0xfe, sizeof(Chunk) );
370 #endif
371  chunk->next = _root;
372  _root = chunk;
373  }
374  void Trace( const char* name ) {
375  printf( "Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d blocks=%d\n",
376  name, _maxAllocs, _maxAllocs*SIZE/1024, _currentAllocs, SIZE, _nAllocs, _blockPtrs.Size() );
377  }
378 
379  void SetTracked() {
380  _nUntracked--;
381  }
382 
383  int Untracked() const {
384  return _nUntracked;
385  }
386 
387  // This number is perf sensitive. 4k seems like a good tradeoff on my machine.
388  // The test file is large, 170k.
389  // Release: VS2010 gcc(no opt)
390  // 1k: 4000
391  // 2k: 4000
392  // 4k: 3900 21000
393  // 16k: 5200
394  // 32k: 4300
395  // 64k: 4000 21000
396  enum { COUNT = (4*1024)/SIZE }; // Some compilers do not accept to use COUNT in private part if COUNT is private
397 
398 private:
399  union Chunk {
400  Chunk* next;
401  char mem[SIZE];
402  };
403  struct Block {
404  Chunk chunk[COUNT];
405  };
406  DynArray< Block*, 10 > _blockPtrs;
407  Chunk* _root;
408 
409  int _currentAllocs;
410  int _nAllocs;
411  int _maxAllocs;
412  int _nUntracked;
413 };
